main.go 
package main

import (
    "context"
    "fmt"
    "log"

    "github.com/mem9-ai/drive9/pkg/client"
)

func main() {
    ctx := context.Background()

    // Create client
    c := client.New("http://localhost:9009", "")

    // Write file
    err := c.Write(ctx, "/data/hello.txt", []byte("Hello, drive9!"))
    if err != nil {
        log.Fatal(err)
    }

    // Read file
    data, err := c.Read(ctx, "/data/hello.txt")
    if err != nil {
        log.Fatal(err)
    }
    fmt.Println(string(data))

    // List directory
    entries, err := c.List(ctx, "/data/")
    if err != nil {
        log.Fatal(err)
    }
    for _, e := range entries {
        fmt.Println(e.Name, e.Size, e.ModTime)
    }

    // Zero-copy link
    err = c.Copy(ctx, "/data/hello.txt", "/backup/hello.txt")
    if err != nil {
        log.Fatal(err)
    }
}

main.rs 
use drive9::Client;

#[tokio::main]
async fn main() -> Result<(), drive9::Drive9Error> {
    // Create client
    let c = Client::new("http://localhost:9009", "");

    // Write file
    c.write("/data/hello.txt", b"Hello, drive9!").await?;

    // Read file
    let data = c.read("/data/hello.txt").await?;
    println!("{}", String::from_utf8_lossy(&data));

    // List directory
    let entries = c.list("/data/").await?;
    for e in entries {
        println!("{} {} {:?}", e.name, e.size, e.mtime);
    }

    // Zero-copy link
    c.copy("/data/hello.txt", "/backup/hello.txt").await?;

    // Stream upload
    let file = tokio::fs::File::open("large.bin").await?;
    c.write_stream("/data/large.bin", file, size).await?;

    Ok(())
}
